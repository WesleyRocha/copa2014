	dataType: 'json',
	type: "GET",
	data: "source="+JSON.stringify({"query":{"match_all":{}}}),
	success: function(data) { console.log('Success')},
	error: function(xhr, message, error) {console.log('Error')},
	complete: function() {console.log('Completed')}
});
